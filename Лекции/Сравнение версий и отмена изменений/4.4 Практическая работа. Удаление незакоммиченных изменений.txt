git restore index.html - убрать сделанные изменения в конкретном файле
git restore --staged imdex.html - убрать файл из индекса
git reset --hard - удалание всех незакомиченных изменений, в том числе тех изменений которые были добавлены в индекс. Выполнить жёсткий сброс внесённых изменений
git clean -f удаление файлов которые не в индексе
git rm --cached config.json - удаляем файл из отслеживания
